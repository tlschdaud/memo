3. try-catch-finally 문

try {
    //예외가 발생할 수도 있는 구간

} catch(발생한 예외 객체) {
   //예외가 발생하면 유입되는 구간
   - 발생시에 적절한 조치
} finally {
   // 예외가 발생하는 발생하지 않든, 항상 실행되는 구간
}
